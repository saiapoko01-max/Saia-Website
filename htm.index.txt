    #results a { display: block; margin: 5px 0; text-decoration: none; color: #333; }
  </style>

<body>
  <h1>Welcome to Saia Website</h1>

  <!-- Search Box -->
  <input type="text" id="searchBox" placeholder="Search content...">

  <!-- Category Buttons -->
  <div>
    <button class="category-btn" onclick="filterCategory('Web Development')">Web Development</button>
    <button class="category-btn" onclick="filterCategory('Programming')">Programming</button>
    <button class="category-btn" onclick="filterCategory('Math')">Math</button>
    <button class="category-btn" onclick="filterCategory('All')">Show All</button>
  </div>

  <!-- Results -->
  <div id="results"></div>

  <script>
    const data = [
      {category: "Web Development", title: "HTML Basics", link: "html.html"},
      {category: "Web Development", title: "CSS Styling", link: "css.html"},
      {category: "Programming", title: "Python Intro", link: "python.html"},
      {category: "Programming", title: "JavaScript Guide", link: "js.html"},
      {category: "Math", title: "Linear Algebra", link: "math.html"}
    ];

    function displayResults(items) {
      document.getElementById("results").innerHTML =
        items.map(r => `<a href="${r.link}">${r.title} (${r.category})</a>`).join("");
    }

    // Initial display
    displayResults(data);

    // Search function
    document.getElementById("searchBox").addEventListener("input", function(e) {
      const query = e.target.value.toLowerCase();
      const results = data.filter(item => item.title.toLowerCase().includes(query));
      displayResults(results);
    });

    // Category filter
    function filterCategory(cat) {
      if (cat === "All") {
        displayResults(data);
      } else {
        const results = data.filter(item => item.category === cat);
        displayResults(results);
      }
    }
  </script>
